‚ö†Ô∏èThe Problem with Hardcoded Secrets

‚ùå Never Do This!
resource "aws_db_instance" "example" {
  password = "MyPassword123"  # DANGER! In Git forever
}

variable "api_key" {
  default = "sk-1234567890"   # NEVER hardcode secrets!
}

Secrets hardcoded in .tf files end up in version control, exposing sensitive data to anyone with repository access.

üîíHashiCorp Vault Integration

Vault provides centralized secret management with encryption, access control, and audit logging.

provider "vault" {
  address = "http://vault-server:8200"
  
  auth_login {
    path = "auth/approle/login"
    parameters = {
      role_id   = var.role_id
      secret_id = var.secret_id
    }
  }
}

data "vault_kv_secret_v2" "db_creds" {
  mount = "secret"
  name  = "database-credentials"
}

resource "aws_db_instance" "example" {
  password = data.vault_kv_secret_v2.db_creds.data["password"]
}
